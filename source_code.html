#include <WiFi.h>
#include <DHT.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <Servo.h>

const char* ssid = "Bharat_$$";        // Replace with your WiFi SSID
const char* password = "Ghazipur@0289$"; // Replace with your WiFi Password

#define DHTPIN 2             // Pin for DHT sensor
#define SOIL_SENSOR_PIN A0   // Pin for soil moisture sensor
#define RELAY_PIN 3          // Pin for relay
#define SERVO_PIN 4          // Pin for servo motor

DHT dht(DHTPIN, DHT22);     // DHT sensor object (DHT22)

WiFiServer server(80);       // Web server on port 80
LiquidCrystal_I2C lcd(0x27, 16, 2); // LCD object with address 0x27 and 16x2 display
Servo servo;                // Servo object

int relayStatus = LOW;  // Initial relay state
int soilMoistureThreshold = 400; // Soil moisture threshold to trigger relay

void setup() {
  Serial.begin(115200);
  
  // Connect to Wi-Fi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(100); // Shorter delay for faster connection
  }
  Serial.println("Connected to Wi-Fi");
  
  // Print the IP address to the Serial Monitor
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP());
  
  server.begin();    // Start the web server
  dht.begin();       // Initialize the DHT sensor
  
  lcd.init();        // Initialize the LCD
  lcd.backlight();   // Turn on the backlight

  pinMode(RELAY_PIN, OUTPUT); // Set relay pin as output
  digitalWrite(RELAY_PIN, LOW); // Ensure relay is initially off
  
  servo.attach(SERVO_PIN); // Attach the servo to the specified pin
  servo.write(0); // Initialize servo position to 0 degrees
}

void loop() {
  // Handle any incoming web clients
  WiFiClient client = server.available();
  if (client) {
    String request = "";
    while (client.connected()) {
      if (client.available()) {
        char c = client.read();
        request += c;
        if (c == '\n') {
          if (request.indexOf("GET /pump/on") >= 0) {
            digitalWrite(RELAY_PIN, HIGH);  // Turn pump ON
            relayStatus = HIGH;
          }
          if (request.indexOf("GET /pump/off") >= 0) {
            digitalWrite(RELAY_PIN, LOW);   // Turn pump OFF
            relayStatus = LOW;
          }
          if (request.indexOf("GET /servo/rotate") >= 0) {
            rotateServo(); // Trigger servo rotation
          }
          break;
        }
      }
    }

    // Serve the web page to the client
    handleClient(client);
  }

  // Continuously read sensor values and control relay without delay
  float humidity = dht.readHumidity();
  float temperature = dht.readTemperature();
  int soilMoisture = analogRead(SOIL_SENSOR_PIN);

  // Check soil moisture and control the relay with no delay
  if (soilMoisture < soilMoistureThreshold && relayStatus == LOW) {
    digitalWrite(RELAY_PIN, HIGH);  // Turn pump ON if soil is dry
    relayStatus = HIGH;
  } else if (soilMoisture >= soilMoistureThreshold && relayStatus == HIGH) {
    digitalWrite(RELAY_PIN, LOW);   // Turn pump OFF if soil is wet
    relayStatus = LOW;
  }

  // Update LCD display with the requested format
  lcd.clear();
  lcd.setCursor(0, 0);   // First line
  lcd.print("T: ");
  lcd.print(temperature, 1);  // Temperature
  lcd.print(" H: ");
  lcd.print(humidity, 1);     // Humidity

  lcd.setCursor(0, 1);   // Second line
  lcd.print("S: ");
  lcd.print(soilMoisture / 1023.0 * 100.0, 1);  // Soil moisture as percentage
  lcd.print(" P: ");
  lcd.print(relayStatus == HIGH ? "ON" : "OFF"); // Pump status

  delay(100);  // Reduced delay for faster web response
}

void handleClient(WiFiClient &client) {
  // Read sensor data
  float temperature = dht.readTemperature();  // Get the latest temperature reading
  float humidity = dht.readHumidity();       // Get the latest humidity reading
  int soilMoisture = analogRead(SOIL_SENSOR_PIN); // Get the latest soil moisture reading

  // Create the response
  String response = "HTTP/1.1 200 OK\r\n";
  response += "Content-Type: text/html; charset=UTF-8\r\n";
  response += "Connection: close\r\n\r\n";  // End of HTTP headers

  response += "";

  response += "
Smart Village - Irrigation System
";  // Title

  response += "
";

  // Card for Sensor Data (System Status)
  response += "
"; 
  response += "
System Status
";
  response += "
💧 Humidity:
";
  response += "
" + String(humidity, 1) + "%
";
  response += "
🌡️ Temperature:
";
  response += "
" + String(temperature, 1) + "°C
";
  response += "
🌱 Soil Moisture:
";
  response += "
" + String((soilMoisture / 1023.0) * 100, 1) + "%
";
  response += "
🚰 Pump Status: " + String(relayStatus == HIGH ? "ON" : "OFF") + "
";
  response += "
";

  response += "
";
  response += "";

  client.print(response);
  client.stop();
}

void rotateServo() {
  for (int i = 0; i < 5; i++) {
    // Rotate from 0 to 90 degrees
    for (int pos = 0; pos <= 90; pos++) {
      servo.write(pos);
    }
    delay(200);  // Wait for 500ms at 90 degrees
    // Rotate from 90 to 0 degrees
    for (int pos = 90; pos >= 0; pos--) {
      servo.write(pos);
    }
    delay(200);  // Wait for 500ms at 0 degrees
  }
}
